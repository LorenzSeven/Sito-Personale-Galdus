	private static void esercizio11()
	{
		System.out.println("Trovare tutte le occorrenze di tutte le parole");
		Scanner input = new Scanner (System.in);
		ArrayList< String > vettoreParole = new ArrayList< String >();
		ArrayList< String > vettoreParoleNoDuplicati = new ArrayList< String >();
		ArrayList< Integer > vettoreOccorrenzeNoDuplicati = new ArrayList< Integer >();
		String parola;
		boolean parolaPresente= false;
		
		do 
		{
			System.out.println("Inserisci una parola nella lista");
			parola = input.next();
			//per i che va da 0 a lunghezza vettoreParole
			for (int i=0;i < vettoreParole.size();i++)
			{
				//se c'è la parola nella listaParole
				if (parola.equals(vettoreParole.get(i)))
				{
					parolaPresente = true;
				}
			} 
			//se la parola non è presente nella lista e non è uguale alla stringa "0"
			//ATTENZIONE che "0" non è lo 0 numero intero. "0" è una stringa.
			if (parolaPresente == false && (!parola.equals("0")))
			{
				System.out.println("Hai inserito una parola nuova");
				vettoreParole.add(parola); //aggiungi la parola alla listaParole
				vettoreParoleNoDuplicati.add(parola); //aggiungo la parola al vettoreParoleNoDuplicati
				vettoreOccorrenzeNoDuplicati.add(1); //aggiungo 1 al vettoreOccorrenzeNoDuplicati
				//ho messo 1 ma poteva essere anche 0 o un altro numero, l'importante è che
				//inserisca un elemento nel vettoreOccorrenzeNoDuplicati, servirà dopo per l'occorrenza
			}
			//se la parola è presente nella listaParole
			if (parolaPresente == true)
			{
				//aggiungi la parola alla listaParole
				vettoreParole.add(parola); 
				
				//calcolo dell'occorrenza della parola nella listaParole
				int occorrenza = 0;
				for (int i=0;i< vettoreParole.size();i++) 
				{								
					//se la parola c'è nella listaParole allora
					if (parola.equals(vettoreParole.get(i))) 
					{
						occorrenza += 1; //aggiungi +1 all'occorrenza
					}
				}	
				System.out.println("La parola inserita è già presente nella lista");
				System.out.println("Ed ha occorrenza: " + occorrenza);
				System.out.println("");		
				
				//cerco l'indice della parola nella lista vettoreParoleNoDuplicati
				int indiceParola = 0;
				for (int i=0;i< vettoreParoleNoDuplicati.size();i++) //uso la j perchè la i è già utilizzata
				{	
					if (parola.equals(vettoreParoleNoDuplicati.get(i)))
					{
						indiceParola = i;
					}
				}
				//stampa dell'indice della parola nella lista vettoreParoleNoDuplicati
				//System.out.println("Indice parola nella lista vettoreParoleNoDuplicati: " + indiceParola);
				
				//sostituisci la vecchia occorrenza della parola con quella nuova
				vettoreOccorrenzeNoDuplicati.set(indiceParola, occorrenza);
				//resetto la parolaPresente
				parolaPresente = false;
			}
		}
		while(!parola.equals("0"));
		
		//stampo la lista senza duplicati
		System.out.println("Lista parole: " + vettoreParoleNoDuplicati);
		System.out.println("Lista occorrenze: " + vettoreOccorrenzeNoDuplicati);
	}
