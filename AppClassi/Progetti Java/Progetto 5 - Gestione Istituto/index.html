import java.util.ArrayList;

public class Classe 
{
    private ArrayList< Studente > listaStudenti;
    
    //Costruttori (Overload)
    public Classe(ArrayList< Studente > listaStudenti)
    {
    	this.listaStudenti = listaStudenti;
    }
    public Classe()
    {
    	this.listaStudenti = new ArrayList< Studente >();
    }
    
    
	public void setListaStudenti(ArrayList< Studente > listaStudenti)
	{
		this.listaStudenti = listaStudenti;
	}
	public ArrayList< Studente > getListaStudenti()
	{
		return listaStudenti;
	}
	
	public boolean aggiungiListaStudenti(ArrayList< Studente > aggiungiListaStudenti)
	{
		return this.listaStudenti.addAll(aggiungiListaStudenti);
	}
	public boolean aggiungiStudente(Studente studente)
	{
		System.out.println("Alla classe si aggiunge lo studente");
		return listaStudenti.add(studente);
	}
	public boolean rimuoviStudente(Studente studente)
	{
		System.out.println("Alla classe si rimuove lo studente");
		return listaStudenti.remove(studente);
	}
}
