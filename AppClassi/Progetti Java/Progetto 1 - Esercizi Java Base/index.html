import java.util.Arrays; 

class AlberoBinario 
{
	public Node root;
	
    public static class Node 
    {
        int item;
        Node left, right;

        //metodo costruttore
        public Node(int key) 
        {
            item = key;
            left = right = null;
        }
    }
    public void insertBalanced(int[] nums) 
    {
        Arrays.sort(nums); // Ordina l'array per semplificare l'inserimento bilanciato
        root = insertBalancedRecursive(nums, 0, nums.length - 1);
    }

    // Metodo ricorsivo per inserire un array di numeri bilanciato
    public static Node insertBalancedRecursive(int[] nums, int start, int end) 
    {
        if (start > end) 
        {
            return null;
        }

        int mid = (start + end) / 2;
        Node newNode = new Node(nums[mid]);

        newNode.left = insertBalancedRecursive(nums, start, mid - 1);
        newNode.right = insertBalancedRecursive(nums, mid + 1, end);

        return newNode;
    }
    public static void inOrder(Node node) //sinistra radice destra
    {
        if (node == null)  {return;}
        inOrder(node.left);
        System.out.print(node.item + " ");
        inOrder(node.right);
    }

    public static void preOrder(Node node) //radice sinistra destra
    {
        if (node == null)  {return;}
        System.out.print(node.item + " ");
        preOrder(node.left);
        preOrder(node.right);
    }

    public static void postOrder(Node node) //sinistra destra radice
    {
        if (node == null)  {return;}
        postOrder(node.left);
        postOrder(node.right);
        System.out.print(node.item + " ");
    }
    public static int maxDepth(Node node) //massima profondita' albero
    {
        if (node == null)  {return 0;} 
        else 
        {
            int leftDepth = maxDepth(node.left);
            int rightDepth = maxDepth(node.right);

            return Math.max(leftDepth, rightDepth) + 1;
        }
    }
    
    // Metodo per verificare se un albero e' un sottoalbero di un altro albero
    public static boolean isSubtree(Node tree1, Node tree2) 
    {
        // Se l'albero da confrontare e' vuoto, e' un sottoalbero
        if (tree2 == null) {return true;}

        // Se l'albero in cui cercare e' vuoto, non puo' contenere un sottoalbero
        if (tree1 == null) {return false;}

        // Verifica se gli alberi hanno la stessa radice e i sottoalberi corrispondenti
        return isIdentical(tree1, tree2) || isSubtree(tree1.left, tree2) || isSubtree(tree1.right, tree2);
    }
    // Metodo ausiliario per verificare l'identita' di due alberi
    private static boolean isIdentical(Node tree1, Node tree2) 
    {
        // Se entrambi gli alberi sono vuoti, sono identici
        if (tree1 == null && tree2 == null) {return true;}

        // Se uno dei due alberi e' vuoto mentre l'altro no, non sono identici
        if (tree1 == null || tree2 == null) {return false;}

        // Verifica la corrispondenza dei nodi e dei sottoalberi
        return (tree1.item == tree2.item) && isIdentical(tree1.left, tree2.left) && isIdentical(tree1.right, tree2.right);
    }
}
