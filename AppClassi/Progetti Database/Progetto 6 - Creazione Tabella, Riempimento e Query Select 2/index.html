--creazione tabella Utente
create table Utente(
ID_UTENTE int not null,
NOME      VARCHAR2(20),
COGNOME   VARCHAR2(20),
DATA_NASCITA   date,
PRIMARY KEY(ID_UTENTE)
);

create table Studente(
NUM_MATRICOLA int not null,
STUDENTE int,
CLASSE VARCHAR2(10),
PRIMARY KEY(NUM_MATRICOLA),
FOREIGN KEY(STUDENTE) REFERENCES Utente(ID_UTENTE)
);

create table Professore(
CODICE_PROFESSORE int not null,
DOCENTE int,
TITOLO_STUDIO VARCHAR2(50),
PRIMARY KEY(CODICE_PROFESSORE),
FOREIGN KEY(DOCENTE) REFERENCES Utente(ID_UTENTE)
);


--non c'è bisogno di inserire una primary key, perchè 
create table Insegna(
DOCENTE int,
MATERIA VARCHAR2(20),
CLASSE  VARCHAR2 (10),
FOREIGN KEY(DOCENTE) REFERENCES Professore(CODICE_PROFESSORE)
);

create table Voto(
ID_VOTO int not null,
MATRICOLA int,
DATA_VOTO date,
MATERIA VARCHAR2 (20),
ARGOMENTO VARCHAR2 (50),
VOTO int,
PRIMARY KEY(ID_VOTO),
FOREIGN KEY(MATRICOLA) REFERENCES Studente(NUM_MATRICOLA)
);
